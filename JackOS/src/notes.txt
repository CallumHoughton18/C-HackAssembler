    function int convertUIntStringToInt(String uintAsString) {
        var int length, i, power, j, multiplier, digit, result;

        let length = uintAsString.length();
        let i = 0;
        let result = 0;

        while (i < length) {
            let multiplier = 1;
            let power = length - i;
            let j = 1;

            while (j < power) {
                let multiplier = multiplier * 10;
            }

            let digit = uintAsString[i] + 48;
            let result = result + digit * multiplier;

            let i = i + 1;
        }

        return result;
    }
